CREATE OR ALTER PROC sp_StartTest
(
    @TestID INT,
    @StudentId INT
)
AS
BEGIN
    SET NOCOUNT ON;

    BEGIN TRY

        DECLARE @CurrentTime DATETIME = GETDATE();

        -- Validation Section
      
        -- Student Exists
        IF NOT EXISTS (
            SELECT 1
            FROM dbo.Users
            WHERE UserId = @StudentId
              AND Role = 'Student'
        )
        BEGIN
            GOTO ValidationFailed;
        END

        -- Student Belongs To Target Class
        IF NOT EXISTS (
            SELECT 1
            FROM dbo.Users u
            INNER JOIN dbo.Tests t
                ON t.TestId = @TestID
            WHERE u.UserId = @StudentId
              AND u.StudentClassId = t.TargetClassId
        )
        BEGIN
            GOTO ValidationFailed;
        END

        -- Test Exists & Active
        IF NOT EXISTS (
            SELECT 1
            FROM dbo.Tests
            WHERE TestId = @TestID
              AND IsActive = 1
        )
        BEGIN
            GOTO ValidationFailed;
        END

        -- Time Window Check 
        IF NOT EXISTS (
            SELECT 1
            FROM dbo.Tests
            WHERE TestId = @TestID
              --AND @CurrentTime BETWEEN StartDateTime AND EndDateTime
        )
        BEGIN
            GOTO ValidationFailed;
        END

        -- Prevent Re-attempt
        IF EXISTS (
            SELECT 1
            FROM dbo.TestResults
            WHERE TestId = @TestID
              AND StudentId = @StudentId
        )
        BEGIN
            GOTO ValidationFailed;
        END

        -- SUCCESS 
        -- Result Set 1: Test Info
        SELECT
            t.TestId,
            t.Title,
            t.Description,
            t.StartDateTime,
            t.EndDateTime,
            t.TotalMarks
        FROM dbo.Tests t
        WHERE t.TestId = @TestID;

        -- Result Set 2: Questions
        SELECT
            q.QuestionId,
            q.QuestionText,
            q.OptionA,
            q.OptionB,
            q.OptionC,
            q.OptionD,
            q.Marks
        FROM dbo.Questions q
        WHERE q.TestId = @TestID;

        -- Result Set 3: Response
        SELECT 1 AS Success,
               'Test started successfully.' AS Message;

        RETURN;

        -- FAILURE FLOW (Always Return 3 Result Sets)
        ValidationFailed:

        -- Empty Test
        SELECT 
            NULL AS TestId,
            NULL AS Title,
            NULL AS Description,
            NULL AS StartDateTime,
            NULL AS EndDateTime,
            NULL AS TotalMarks
        WHERE 1 = 0;

        -- Empty Questions
        SELECT 
            NULL AS QuestionId,
            NULL AS QuestionText,
            NULL AS OptionA,
            NULL AS OptionB,
            NULL AS OptionC,
            NULL AS OptionD,
            NULL AS Marks
        WHERE 1 = 0;

        -- Failure Response
        SELECT 0 AS Success,
               'Test cannot be started.' AS Message;

    END TRY
    BEGIN CATCH

        -- Always return 3 result sets even on SQL error
        SELECT 
            NULL AS TestId,
            NULL AS Title,
            NULL AS Description,
            NULL AS StartDateTime,
            NULL AS EndDateTime,
            NULL AS TotalMarks
        WHERE 1 = 0;

        SELECT 
            NULL AS QuestionId,
            NULL AS QuestionText,
            NULL AS OptionA,
            NULL AS OptionB,
            NULL AS OptionC,
            NULL AS OptionD,
            NULL AS Marks
        WHERE 1 = 0;

        SELECT 0 AS Success,
               ERROR_MESSAGE() AS Message;

    END CATCH
END
GO
SELECT *
FROM dbo.Tests; -- TestID 1002, TargetClassId 1
GO
SELECT UserId, Role, StudentClassId
FROM dbo.Users
WHERE Role = 'Student'; -- StudentId 5 since he is in TargetClassId 1
GO
EXEC sp_StartTest
    @TestID = 1002,
    @StudentId = 4;
